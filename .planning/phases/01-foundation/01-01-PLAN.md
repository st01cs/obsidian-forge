Create `src/VaultAdapter.ts` that wraps all vault operations through app.vault API. Per the research, VaultAdapter is the foundation that ALL tools depend on — never call app.vault directly.

```typescript
import { App, TFile, TFolder, CachedMetadata, normalizePath } from 'obsidian';

export interface NoteMetadata {
  path: string;
  frontmatter: Record<string, any> | null;
  links: Array<{ link: string; displayText: string }>;
  outlinks: Array<{ link: string; displayText: string }>;
  backlinks: Array<{ link: string; displayText: string }>;
  tags: string[];
}

export class VaultAdapter {
  constructor(private app: App) {}

  /**
   * Normalize a path to use forward slashes consistently.
   */
  private normalizePath(path: string): string {
    return normalizePath(path);
  }

  /**
   * Get a TFile by path, throwing if not found or not a file.
   */
  private getFile(path: string): TFile {
    const file = this.app.vault.getAbstractFileByPath(this.normalizePath(path));
    if (!file) throw new Error(`File not found: ${path}`);
    if (!(file instanceof TFile)) throw new Error(`Not a file: ${path}`);
    return file;
  }

  /**
   * Read a note's content using cachedRead (preferred over read for performance).
   * VAULT-01
   */
  async readNote(path: string): Promise<string> {
    const file = this.getFile(path);
    return await this.app.vault.cachedRead(file);
  }

  /**
   * Write content to a note. Creates if not exists, modifies if exists.
   * VAULT-02
   */
  async writeNote(path: string, content: string): Promise<void> {
    const normalizedPath = this.normalizePath(path);
    const file = this.app.vault.getAbstractFileByPath(normalizedPath);
    if (file instanceof TFile) {
      await this.app.vault.modify(file, content);
    } else {
      await this.app.vault.create(normalizedPath, content);
    }
  }

  /**
   * Atomic edit: read current content, apply transform function, write result.
   * Uses vault.process() for atomic operation.
   * VAULT-02
   */
  async editNote(path: string, fn: (content: string) => string): Promise<void> {
    const file = this.getFile(path);
    await this.app.vault.process(file, fn);
  }

  /**
   * Get file metadata (frontmatter, links, tags) without reading content.
   * VAULT-01
   */
  getMetadata(path: string): NoteMetadata | null {
    const file = this.getFile(path);
    const cache = this.app.metadataCache.getFileCache(file);
    if (!cache) return null;
    return {
      path: file.path,
      frontmatter: cache.frontmatter ?? null,
      links: (cache.links ?? []).map(l => ({ link: l.link, displayText: l.displayText })),
      outlinks: (cache.outgoingLinks ?? []).map(l => ({ link: l.link, displayText: l.displayText })),
      backlinks: this.getBacklinksForFile(file.path),
      tags: cache.tags ?? []
    };
  }

  /**
   * Get all files in vault matching an optional extension filter.
   * Used by list_files tool and vault_search.
   */
  listFiles(extension?: string): TFile[] {
    const files = this.app.vault.getFiles();
    if (extension) {
      return files.filter(f => f.extension === extension);
    }
    return files;
  }

  /**
   * Get backlinks pointing TO a specific file.
   * Uses metadataCache.resolvedLinks.
   */
  getBacklinksForFile(path: string): Array<{ link: string; displayText: string }> {
    const resolvedLinks = this.app.metadataCache.resolvedLinks ?? {};
    const backlinks: Array<{ link: string; displayText: string }> = [];
    const normalizedPath = this.normalizePath(path);
    for (const [sourcePath, links] of Object.entries(resolvedLinks)) {
      for (const [target, count] of Object.entries(links)) {
        if (target === normalizedPath) {
          backlinks.push({ link: sourcePath, displayText: sourcePath });
        }
      }
    }
    return backlinks;
  }

  /**
   * Search notes by filename (simple contains).
   * Content search done by agent using vault_read + grep pattern.
   */
  searchByFilename(query: string): TFile[] {
    const files = this.listFiles('md');
    const lowerQuery = query.toLowerCase();
    return files.filter(f => f.name.toLowerCase().includes(lowerQuery));
  }

  /**
   * Rename a note. Returns the new path.
   */
  async renameNote(oldPath: string, newPath: string): Promise<string> {
    const file = this.getFile(oldPath);
    const normalizedNewPath = this.normalizePath(newPath);
    await this.app.vault.rename(file, normalizedNewPath);
    return normalizedNewPath;
  }

  /**
   * Create a folder if it doesn't exist.
   */
  async createFolder(path: string): Promise<void> {
    const normalizedPath = this.normalizePath(path);
    const existing = this.app.vault.getAbstractFileByPath(normalizedPath);
    if (existing instanceof TFolder) return; // Already exists
    await this.app.vault.createFolder(normalizedPath);
  }

  /**
   * Check if a path exists in vault.
   */
  exists(path: string): boolean {
    const file = this.app.vault.getAbstractFileByPath(this.normalizePath(path));
    return file !== null;
  }
}
```

Key patterns per research:
- Use `cachedRead()` not `read()` for performance
- Use `vault.process()` for atomic edits
- `normalizePath()` for consistent path handling
- `getFile()` helper to validate TFile type
- Constructor injection of `App` instance
